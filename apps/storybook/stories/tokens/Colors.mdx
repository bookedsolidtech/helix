export const allSubgroups = getColorSubgroups(tokenEntries);
export const colorStats = getTokenStats(tokenEntries.filter((t) => t.category === 'color'));

export const semanticRampNames = ['success', 'warning', 'error', 'info'];
export const brandRampNames = ['primary', 'secondary', 'accent'];

export function resolveHex(token) {
  const raw = token.value;
  if (isHexColor(raw)) return raw;
  return resolveTokenRef(raw, tokenMap);
}

export function findSubgroup(label) {
  return allSubgroups.find((sg) => sg.label.toLowerCase() === label.toLowerCase());
}

export function copyToClipboard(text) {
  navigator.clipboard.writeText(text).catch(() => {});
}

</div>
<div className="hx-colors-grid">

const contrast = isHexColor(hex) ? getContrastColor(hex) : '#000000';
return (
<div key={token.name} className="hx-colors-swatch">

</div>
<div className="hx-colors-swatch-info">
<button
className="hx-colors-swatch-name"
title={`Click to copy: ${token.name}`}

{token.name.replace('--hx-color-', '')}
</button>
<span className="hx-colors-swatch-value">{token.name}</span>
</div>
</div>
);
})}
</div>
</div>
);
})}

## Overview

The `hx-select` component is a styled wrapper around the native `<select>` element, providing consistent
visual design and behavior across the HELIX design system. It uses the Form-Associated Custom Elements API
via ElementInternals, ensuring full compatibility with standard HTML forms and validation.

The component provides three size variants (`sm`, `md`, `lg`), validation states (required, error), and
helper elements (label, help text, error messages). All interactive states (hover, focus, disabled) are
fully styled and accessible.

### When to Use

- **Department selection** — Choose departments, wards, or clinical units
- **Blood type, priority, or status** — Pick from a fixed set of categorical values
- **Insurance provider selection** — Select from a known list of payers
- **Medication routes** — Choose administration routes (oral, IV, IM, etc.)
- **Diagnosis code selection** — Pick ICD-10 codes from a long list

### When Not to Use

- **Free text input** — Use `hx-text-input` for arbitrary text entry
- **Multi-select** — Use `hx-checkbox` or a custom multi-select component
- **Very long lists (500+ options)** — Consider an autocomplete or search pattern
- **Boolean choices** — Use `hx-switch` or `hx-radio-group` for yes/no or on/off

## Accessibility

The Select component follows WCAG 2.1 AA guidelines:

- **Keyboard navigation** — Full support for arrow keys, Enter, Space, Home, End
- **Screen reader support** — Proper labels, ARIA attributes, and announcements
- **Focus visible** — Clear focus ring for keyboard users
- **Error announcements** — Error messages are associated with the select via `aria-describedby`
- **Required field indicator** — Visual and semantic indication of required fields

## Properties

| Property      | Type                   | Default | Description                                           |
| ------------- | ---------------------- | ------- | ----------------------------------------------------- |
| `label`       | `string`               | `''`    | Visible label text for the select                     |
| `placeholder` | `string`               | `''`    | Placeholder text shown as the first disabled option   |
| `value`       | `string`               | `''`    | Current value of the select                           |
| `hx-size`     | `'sm' \| 'md' \| 'lg'` | `'md'`  | Size variant of the select                            |
| `required`    | `boolean`              | `false` | Whether the select is required for form submission    |
| `disabled`    | `boolean`              | `false` | Whether the select is disabled                        |
| `error`       | `string`               | `''`    | Error message to display (puts select in error state) |
| `help-text`   | `string`               | `''`    | Help text displayed below the select                  |
| `name`        | `string`               | `''`    | Name attribute for form submission                    |

## Form Participation

The `hx-select` component uses ElementInternals to participate in native form submission:

- **FormData integration** — The `name` and `value` are included in `FormData` on form submit
- **Validation** — The `required` attribute triggers native HTML5 validation
- **Reset support** — The select resets to its initial value when the form is reset

## Design Guidelines

### Label Guidelines

- Always provide a clear, concise label
- Use sentence case (e.g., "Primary insurance provider")
- Avoid redundant text (don't repeat "Select" in the label if the placeholder says "Select...")

### Option Text

- Keep option text short (1-5 words when possible)
- Use parallel structure (all nouns, all verb phrases, etc.)
- Alphabetize unless there's a logical order (STAT → Urgent → Routine)

### Placeholder Text

- Use descriptive placeholders ("Select department..." not "Choose one")
- Avoid placeholder text that duplicates the label

### Validation

- Always set `required` for mandatory fields
- Provide clear error messages that tell the user how to fix the issue
- Use `help-text` to provide guidance before errors occur

```php
/**
 * Validation handler for patient form.
 */
function mymodule_patient_form_validate($form, &$form_state) {
  $department = $form_state->getValue('department');

  // Validate department
  if (empty($department)) {
    $form_state->setErrorByName('department', t('Department is required for admissions.'));
  }

  $blood_type = $form_state->getValue('blood_type');

  // Cross-field validation
  if ($department === 'oncology' && empty($blood_type)) {
    $form_state->setErrorByName('blood_type', t('Blood type is required for oncology admissions.'));
  }
}
```

### Drupal Behaviors for Event Handling

```javascript
/**
 * @file
 * hx-select Drupal behavior.
 */

(function (Drupal, once) {
  'use strict';

  /**
   * Attach hx-select event listeners.
   */
  Drupal.behaviors.hxSelect = {
    attach: function (context, settings) {
      // Initialize all selects
      once('hx-select-init', 'hx-select', context).forEach((select) => {
        // Listen for value changes
        select.addEventListener('hx-change', (e) => {
          const { value } = e.detail;
          const name = select.getAttribute('name');

          console.log(`Select ${name} changed to: ${value}`);

          // Example: Conditional field display based on selection
          if (name === 'department') {
            handleDepartmentChange(value, context);
          }

          // Example: Trigger AJAX refresh of dependent field
          if (name === 'insurance_provider') {
            refreshPlanOptions(value);
          }
        });

        // Optional: Track analytics
        select.addEventListener('hx-change', (e) => {
          if (typeof gtag !== 'undefined') {
            gtag('event', 'select_change', {
              field_name: select.getAttribute('name'),
              selected_value: e.detail.value,
            });
          }
        });
      });
    },

    detach: function (context, settings, trigger) {
      if (trigger === 'unload') {
        context.querySelectorAll('hx-select').forEach((select) => {
          select.removeEventListener('hx-change', () => {});
        });
      }
    },
  };

  /**
   * Handle department selection changes.
   */
  function handleDepartmentChange(department, context) {
    const specialtyField = context.querySelector('[name="specialty"]');

    if (specialtyField) {
      // Show/hide specialty field based on department
      if (department === 'cardiology' || department === 'neurology') {
        specialtyField.closest('.form-item').style.display = 'block';
      } else {
        specialtyField.closest('.form-item').style.display = 'none';
      }
    }
  }

  /**
   * Refresh dependent select options via AJAX.
   */
  function refreshPlanOptions(providerId) {
    const planSelect = document.querySelector('hx-select[name="insurance_plan"]');

    if (!planSelect) return;

    // Show loading state
    planSelect.setAttribute('disabled', 'true');

    // Fetch updated options
    fetch(`/api/insurance-plans/${providerId}`)
      .then((response) => response.json())
      .then((data) => {
        // Clear existing options
        planSelect.innerHTML = '<option value="">- Select plan -</option>';

        // Add new options
        data.plans.forEach((plan) => {
          const option = document.createElement('option');
          option.value = plan.id;
          option.textContent = plan.name;
          planSelect.appendChild(option);
        });

        // Re-enable select
        planSelect.removeAttribute('disabled');
      })
      .catch((error) => {
        console.error('Failed to load insurance plans:', error);
        planSelect.removeAttribute('disabled');
      });
  }
})(Drupal, once);
```

### Validation Error Display

**Displaying server-side validation errors:**

```javascript
(function (Drupal, once) {
  Drupal.behaviors.hxSelectValidation = {
    attach: function (context) {
      // Process server-side validation errors
      once('hx-select-errors', 'hx-select', context).forEach((select) => {
        const name = select.getAttribute('name');
        const errorMessage = drupalSettings.validationErrors?.[name];

        if (errorMessage) {
          select.setAttribute('error', errorMessage);
        }
      });

      // Clear error on user interaction
      once('hx-select-error-clear', 'hx-select[error]', context).forEach((select) => {
        select.addEventListener('hx-change', () => {
          select.removeAttribute('error');
        });
      });
    },
  };
})(Drupal, once);
```
